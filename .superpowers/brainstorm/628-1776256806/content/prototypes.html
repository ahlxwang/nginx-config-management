upstream.conf 生产集群
# upstream 配置
upstream backend {
  server 192.168.1.10:8080;
  server 192.168.1.11:8080;
  keepalive 32;
}
 
server {
  listen 80;
  server_name example.com;
  location / {
    proxy_pass http://backend;
  }
}
